87 std::vector<detail::kernel> create_kernel(const std::vector<command_queue> &queues, const std::string &file, const std::string &kernel_name) {
88  // read kernel
89  std::string kernel_src_string;
90  {
91  std::ifstream in{ file };
92  if (in.fail()) {
93  throw backend_exception{ fmt::format("Couldn't open kernel source file: {}!", file) };
94  }
95  in.seekg(0, std::ios::end);
96  std::streamsize len = in.tellg();
97  in.seekg(0, std::ios::beg);
98 
99  kernel_src_string.resize(len);
100  in.read(kernel_src_string.data(), len);
101  }
102 
103  // replace type
104  ::plssvm::detail::replace_all(kernel_src_string, "real_type", ::plssvm::detail::arithmetic_type_name<real_type>());
105  ::plssvm::detail::replace_all(kernel_src_string, "size_type", ::plssvm::detail::arithmetic_type_name<size_type>());
106  // replace constants
107  ::plssvm::detail::replace_all(kernel_src_string, "INTERNAL_BLOCK_SIZE", fmt::format("{}", INTERNAL_BLOCK_SIZE));
108  ::plssvm::detail::replace_all(kernel_src_string, "THREAD_BLOCK_SIZE", fmt::format("{}", THREAD_BLOCK_SIZE));
109 
110  error_code err;
111 
112  // create program
113  const char *kernel_src_ptr = kernel_src_string.c_str();
114  // TODO: not all command queue must have the same context (but this would be highly unlikely)
115  cl_program program = clCreateProgramWithSource(queues[0].context, 1, &kernel_src_ptr, nullptr, &err);
116  if (!err) {
117  throw backend_exception{ fmt::format("Error creating OpenCL program ({})!", err) };
118  }
119  // TODO: add optimization flags?
120  err = clBuildProgram(program, 0, nullptr, nullptr, nullptr, nullptr);
121  if (!err) {
122  // TODO: c++-ify
123  // Determine the size of the log
124  size_t log_size;
125  clGetProgramBuildInfo(program, queues[0].device, CL_PROGRAM_BUILD_LOG, 0, NULL, &log_size);
126 
127  // Allocate memory for the log
128  char *log = (char *) malloc(log_size);
129 
130  // Get the log
131  clGetProgramBuildInfo(program, queues[0].device, CL_PROGRAM_BUILD_LOG, log_size, log, NULL);
132 
133  // Print the log
134  throw backend_exception{ fmt::format("Error building OpenCL program ({})!: {}\n", err, log) };
135  }
136 
137  // build kernels
138  std::vector<detail::kernel> kernels;
139  for ([[maybe_unused]] const command_queue &q : queues) {
140  // create kernel
141  kernels.emplace_back(clCreateKernel(program, kernel_name.c_str(), &err));
142  if (!err) {
143  throw backend_exception{ fmt::format("Error creating OpenCL kernel ({})!", err) };
144  }
145  }
146 
147  // release resource
148  if (program) {
149  err = clReleaseProgram(program);
150  if (!err) {
151  throw backend_exception{ fmt::format("Error releasing OpenCL program resources ({})!", err) };
152  }
153  }
154 
155  return kernels;
156 }

186 template <typename... Args>
187 void run_kernel(const command_queue &queue, cl_kernel kernel, const std::vector<std::size_t> &grid_size, const std::vector<std::size_t> &block_size, Args &&...args) {
188  PLSSVM_ASSERT(grid_size.size() == block_size.size(), "grid_size and block_size must have the same number of dimensions!: {} != {}", grid_size.size(), block_size.size());
189  PLSSVM_ASSERT(grid_size.size() <= 3, "The number of dimensions must be less or equal than 3!: {} > 3", grid_size.size());
190 
191  // set kernel arguments
192  set_kernel_args(kernel, std::forward<Args>(args)...);
193 
194  // enqueue kernel in command queue
195  error_code err = clEnqueueNDRangeKernel(queue.queue, kernel, static_cast<cl_int>(grid_size.size()), nullptr, grid_size.data(), block_size.data(), 0, nullptr, nullptr);
196  if (!err) {
197  throw backend_exception{ fmt::format("Error enqueuing OpenCL kernel ({})!", err) };
198  }
199 
200  // wait until kernel computation finished
201  err = clFinish(queue.queue);
202  if (!err) {
203  throw backend_exception{ fmt::format("Error running OpenCL kernel ({})!", err) };
204  }
205 }
