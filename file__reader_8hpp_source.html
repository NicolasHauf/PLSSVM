103 #if defined(PLSSVM_HAS_MEMORY_MAPPING)
104  void open_memory_mapped_file(const std::string_view filename) {
105  // open the file
106  file_descriptor_ = open(filename.data(), O_RDONLY);
107  struct stat attr {};
108  // check if file could be opened
109  if (fstat(file_descriptor_, &attr) == -1) {
110  close(file_descriptor_);
111  throw file_not_found_exception{ fmt::format("Couldn't find file: '{}'!", filename) };
112  }
113  if (attr.st_size == 0) {
114  // can't memory map empty file
115  close(file_descriptor_);
116  this->open_file(filename);
117  } else {
118  // memory map file
119  file_content_ = static_cast<char *>(mmap(nullptr, attr.st_size, PROT_READ, MAP_SHARED, file_descriptor_, 0));
120  // check if memory mapping was successful
121  if (static_cast<void *>(file_content_) == MAP_FAILED) {
122  // memory mapping wasn't successful -> try reading file with std::ifstream
123  close(file_descriptor_);

125  this->open_file(filename);
126  } else {

155  * Split the file into its lines, ignoring empty lines and lines starting with a comment.
156  */
157  void parse_lines(const char comment) {
158  // create view from buffer
159  std::string_view file_content_view{ file_content_, num_bytes_ };
160  std::size_t pos = 0;
161  while (true) {
162  // find newline
163  std::size_t next_pos = file_content_view.find_first_of('\n', pos);
164  if (next_pos == std::string_view::npos) {
165  break;
166  }
167  // remove trailing whitespaces
168  std::string_view sv = trim_left(std::string_view{ file_content_view.data() + pos, next_pos - pos });
169  // add line iff the line is not empty and doesn't with a comment
170  if (!sv.empty() && !starts_with(sv, comment)) {
171  lines_.push_back(sv);
172  }
173  pos = next_pos + 1;
174  }
175  // add last line
176  std::string_view sv = trim_left(std::string_view{ file_content_view.data() + pos, file_content_view.size() - pos });
177  if (!sv.empty() && !starts_with(sv, comment)) {
178  lines_.push_back(sv);
179  }
180  }
