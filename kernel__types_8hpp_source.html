94 template <kernel_type kernel, typename real_type, typename... Args>
95 real_type kernel_function(const std::vector<real_type> &xi, const std::vector<real_type> &xj, Args &&...args) {
96  using namespace plssvm::operators;
97 
98  PLSSVM_ASSERT(xi.size() == xj.size(), "Sizes mismatch!: {} != {}", xi.size(), xj.size());
99 
100  if constexpr (kernel == kernel_type::linear) {
101  static_assert(sizeof...(args) == 0, "Illegal number of additional parameters! Must be 0.");
102  return transposed{ xi } * xj;
103  } else if constexpr (kernel == kernel_type::polynomial) {
104  static_assert(sizeof...(args) == 3, "Illegal number of additional parameters! Must be 3.");
105  const auto degree = static_cast<int>(detail::get<0>(args...));
106  const auto gamma = static_cast<real_type>(detail::get<1>(args...));
107  const auto coef0 = static_cast<real_type>(detail::get<2>(args...));
108  return std::pow(std::fma(gamma, (transposed<real_type>{ xi } * xj), coef0), static_cast<real_type>(degree));
109  } else if constexpr (kernel == kernel_type::rbf) {
110  static_assert(sizeof...(args) == 1, "Illegal number of additional parameters! Must be 1.");
111  const auto gamma = static_cast<real_type>(detail::get<0>(args...));
112  return std::exp(-gamma * squared_euclidean_dist(xi, xj));
113  } else {
114  static_assert(detail::always_false_v<real_type>, "Unknown kernel type!");
115  }
116 }
